#include <iostream>

using namespace std;
void g(int r1, int r2) {
	r1 += r2;
}

void iswap_r(int &r1, int&r2)
{

	int temp{ r1 };
	r1 = r2, r2 = temp;
}
void func(int &r1, int &r2) {
	iswap_r(r1, r2); // call by reference,
	++r1, --r2;
	g(r1, r2); // call by value

}

int main() {
	int x = 10, y = 6;

	func(x, y);
	
	cout << "x = " << x << "y = " << y << endl;
		
	system("pause");
}

#include <iostream>
using namespace std;

struct Date {
	int md, mm, my;
};

Date &setDate(Date &date,int d, int m, int y) {
	date.md = d;
	date.mm = m;
	date.my = y;
	return date;
}
void display(const Date &d) {
	cout << d.md << "/" << d.mm << "/" << d.my << endl;
}
int main() {
	Date d;

	display(setDate(d, 17, 8, 1996));
	getchar();// to prevent close console
}
